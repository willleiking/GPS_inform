// 🎯 Google Places Autocomplete 智慧聯想（內建每月 9000 次自動熔斷熔斷保護） async function initGoogleAutocomplete() { const input = document.getElementById('geoSearchInput'); // 1. 先去 Supabase 檢查本月累計用量 const currentMonthStr = new Date().toISOString().slice(0, 7); // 取得目前年月，例如 "2026-06" let { data: usageData, error } = await window.dbClient .from('google_api_usage') .select('search_count') .eq('current_month', currentMonthStr) .single(); // 如果是新月份，資料庫可能還沒該月資料，手動防呆建立一筆 if (!usageData && !error) { await window.dbClient.from('google_api_usage').insert({ current_month: currentMonthStr, search_count: 0 }); usageData = { search_count: 0 }; } const currentCount = usageData ? usageData.search_count : 0; // 2. ⚡ 安全熔斷攔截：若超過 9,000 次，直接停用輸入框，不載入 Google 功能 if (currentCount >= 9000) { input.disabled = true; input.placeholder = "🛑 本月地圖搜尋額度已達安全上限，暫停使用。"; showToast("⚠️ 系統提示：本月 Google 搜尋已達 9,000 次安全水位，已自動熔斷以防卡片刷爆。", "error"); return; } // 3. 額度安全，正常初始化 Google 下拉聯想 const options = { componentRestrictions: { country: 'tw' }, fields: ['geometry', 'name', 'formatted_address'], types: ['establishment', 'geocode'] }; const autocomplete = new google.maps.places.Autocomplete(input, options); // 當使用者點選下拉選單的地點時 autocomplete.addListener('place_changed', async function() { const place = autocomplete.getPlace(); if (!place.geometry || !place.geometry.location) { showToast('📭 無法取得該地點的經緯度座標', 'error'); return; } const lat = place.geometry.location.lat(); const lng = place.geometry.location.lng(); // 帶入前端表單 document.getElementById('tLat').value = lat.toFixed(6); document.getElementById('tLng').value = lng.toFixed(6); document.getElementById('tName').value = place.name || place.formatted_address; if (radarMap) { radarMap.setView([lat, lng], 16); L.popup().setLatLng([lat, lng]).setContent(`🎯 經由 Google 精準定位：
${place.name}`).openOn(radarMap); } showToast('🎯 Google 定位轉換成功！', 'success'); // 4. 📈 扣額度記帳：成功調用一次 Google，將 Supabase 计数器 +1 // 使用 RPC 或直接更新（這裡採用標準 RPC 或簡易更新，若高併發可用 RPC，此處用簡單的更新確保安全） await window.dbClient.rpc('increment_api_count', { month_str: currentMonthStr }); }); }
